切换R版本
/etc/rstudio/rserver.conf文件储存Rstudio server的配置R版本的选项

sudo nano /etc/rstudio/rserver.conf

# 在rserver.conf文件中写入
rsession-which-r=/home/xujialiu_wsl/anaconda3/envs/r-4.2.3
# 重启rstudio server
sudo rstudio-server restart
